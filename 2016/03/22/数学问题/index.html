#include <stdio.h>
#include <stdlib.h>
//Description:判断square是否满足对称性质
int judgeSymmetry(int square)
{
    int i, j, flag, arr[10];
    //将square每一位放入数组中
    for(i = 0; square; i ++, square /= 10){
        arr[i] = square % 10;
    }
    //判断数组是否符合对称性质
    for(flag = 1, j = i / 2; j >= 0; j --){
        if(arr[j] != arr[i - 1 - j]){
            flag = 0;
            break;
        }
    }
    return flag;
}
int main(){
    int i;
    for(i = 0; i < 256; i ++){
        if(i == 0){
            printf("0\n");
        }else{
            if(judgeSymmetry(i * i))
                printf("%d\n", i);
        }
    }
    return 0;
}

题目：输入两个正整数，求其最大公约数
分析：一般用gcd(a,b)来表示a和b的最大公约数，求解最大公约数用辗转相除法。
辗转相除法：gcd(a,b)=gcd(b,a%b)
